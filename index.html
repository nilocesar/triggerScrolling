triggerScrolling.js

Event trigger on screen elements during scrolling

Download ZIP

TriggerScrolling serves to trigger events to get pre certain elements in the HTML page. Often when we are creating a site, we need to trigger events or modify something, when we get determines part of the site, thinking it was created triggerScrolling, it facilitates this interaction, monitoring the browser scroll, being possible to know in which part exactly site you are using callbacks available.

Implementation

            
<!DOCTYPE html>
<html lang="en">
	<head>
		<meta charset="UTF-8">
		<title>Title page</title>
	</head>

	<body>

        <!-- Body code... -->

		<script src="{{PATH}}/jquery.triggerScrolling-*.min.js"></script>
        <script>
            // Initializing the library
            var trgScrolling = new triggerScrolling();
            
            // Starting plugin when the page is loaded
            $(document).ready(function() {
                trgScrolling.init(function(objCallback) {
                    console.log(objCallback);
                });
            });

            // Running the plugin with the page scroll
            $(window).on('scroll', function(event) {
                trgScrolling.init(function(objCallback) {
                    console.log(objCallback);
                });
            });
        </script>
	</body>
</html>

Callback object

The returned object in the callback will come with the following information: 

            
{
    'page': {
        'first': {
            'identifier': Name alias the element,
            'element': Object javascript the element,
            'top': Position of the scroll in which this element,
            'height': Height size of first element defined with the attribute
        },
        'last': {
            'identifier': Name alias the element,
            'element': Object javascript the element,
            'top': Position of the scroll in which this element,
            'height': Size height of last element set with the attribute
        },
        'bottom': If the scroll arrived at the bottom,
        'top': If the scroll arrived at the top of the page,
    },
    'controls': {
        'prev': {
            // Who is the previous element
            'identifier': Name alias the element,
            'element': Object javascript the element,
            'top': Position of the scroll in which this element,
            'height': Height size of the element defined with attribute
        },
        'current': {
            // Current element
            'identifier': Name alias the element,
            'element': Object javascript the element,
            'top': Position of the scroll in which this element,
            'height': Height size of the element defined with attribute
        },
        'next': {
            // Who is the next element
            'identifier': Name alias the element,
            'element': Object javascript the element,
            'top': Position of the scroll in which this element,
            'height': Height size of the element defined with attribute
        },
    }
};
